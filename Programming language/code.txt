def s1(s):
    c =0
    for i in range(len(s)):
        c+=int(s[i:i+1])
    return c
def answ(n,k):
    c =0
    for i in range(10**k):
        s =str(i)
        if s1(s)==n:
            c+=1
    return c
n ,k=map(int,input().split())
while n!=0:
    print(answ(n,k))
    n ,k=map(int,input().split())
